『编译原理实验』简单decaf词法分析器的Java实现

所以直到最后都不会用eclipse。。。所以真的佩服你院不学Java直接上实验的勇气。。。 Lexer.java [cc lang=”java” tab_size=”2” lines=”40”]

package lexer;

import java.io.IOException; 
import java.util.Hashtable;
public class Lexer {


    public static int line = 1;/*line变量为当前行数*/
    char peek = ' ';/*peek为当前编译的字符*/
    Hashtable words= new Hashtable();


    int readInt() throws IOException{ /*readint方法用来读取int类型的数据*/
        int v=0;
        readch();
        do {
            v = 10 * v + Character.digit(peek, 10);/*将读取的数字累加到和V上*/
            readch();
        } while (Character.isDigit(peek));
        return v;
    }


    Word readStr() throws IOException {/*识别字符串常量。并且在当找不到引号就出现换行的时候输出错误提示以及发生错误的行号*/
        String b="";
        readch();
        while(peek!='"'){
            if(peek=='\n'){
                System.out.println("[错误] 行"+line+": "+"没有找到右引号！ ");
                break;
            }
            b=b+peek;
            readch();
        }
        Word w = new Word(b,Tag.STR);
        readch();
        return w;
    }


    void reserve(Word w) {/*定义reserve方法，用来记录保留字*/
        words.put(w.lexeme, w);
    }


    /*构造函数，初始化所有的tag*/
    public Lexer() {
        reserve(new Word("if", Tag.IF));
        reserve(new Word("else", Tag.ELSE));
        reserve(new Word("while", Tag.WHILE));
        reserve(new Word("do", Tag.DO));
        reserve(new Word("break", Tag.BREAK));
        reserve(Word.True);
        reserve(Word.False);
         /* 新增添的关键字 */ 
        reserve(new Word("class", Tag.CLASS));
        reserve(new Word("void", Tag.VOID));
        reserve(new Word("for", Tag.FOR));
        reserve(new Word("int", Tag.INT));
        reserve(new Word("double", Tag.DOU));
        reserve(new Word("bool", Tag.BOOL));
        reserve(new Word("string", Tag.STRING));
        reserve(new Word("null", Tag.NULL));
        reserve(new Word("this", Tag.THIS));
        reserve(new Word("extends", Tag.EXTENDS));
        reserve(new Word("return", Tag.RETURN));
        reserve(new Word("new", Tag.NEW));
        reserve(new Word("NewArray", Tag.NEWARRAY));
        reserve(new Word("Print", Tag.PRINT));
        reserve(new Word("ReadInteger", Tag.READINT));
        reserve(new Word("ReadLine", Tag.READLINE));
        reserve(new Word("static", Tag.STATIC));
    }


    /*把下一个输入字符读入到变量peek之中*/
    public void readch() throws IOException {
        peek = (char) System.in.read();
        if (peek=='@') {
            System.out.println("[错误]行:"+line+" 禁止在注释以外的地方出现@");/*如果读到不合法的@字符就输出错误信息*/
         readch();
        }
    }


    /*为了识别复合的词法单元，readch方法用来预读取下一个字符*/
    boolean readch(char c) throws IOException {
        peek = (char) System.in.read();
        if (peek != c) {
            return false;
        }
        peek = ' ';
        return true;
    }



    Token Annotation() throws IOException { //用于判断是否有注释的函数
        if(peek=='/') {/*忽略单行注释*/
            for (;; peek = (char) System.in.read()) { /*忽略符号后的所有内容*/
                if (peek == '\n'){
                    line +=1;
                    peek =' ';
                    return new Word("",Tag.ANNOTATION);
                }
            }
        }
        else if(peek=='*') { /*忽略多行注释*/
            for (;; peek = (char) System.in.read()) { /*忽略符号间的所有内容*/
                if (peek == '\n')
                    line +=1;
                if(peek == '*'&&readch('/')) {
                    return new Word("", Tag.ANNOTATION);
                }
            }
        }
        else return new Token('/');
    }

    /*核心检测的scan函数*/
    public Token scan() throws IOException {    
        for (;; readch()) {
            if (peek == ' ' || peek == '\t')
                continue;
            else if (peek == '\n') {
                line += 1;
            } else {
                break;/*略过空白字符*/
            }
        }

        switch (peek) {
        case '&':
           if (readch('&'))  /*识别复合词法单元*/
                return Word.and;
            else
                return new Token('&');
        case '|':
            if (readch('|'))
                return Word.or;
            else
                return new Token('|');
        case '=':
            if (readch('='))
                return Word.eq;
            else
                return new Token('=');
        case '!':
            if (readch('='))
                return Word.ne;
            else
                return new Token('!');
        case '<':
            if (readch('='))
                return Word.le;
            else
                return new Token('<');
        case '>':
            if (readch('='))
                return Word.ge;
            else
                return new Token('>');
        case '/':
            return         Annotation();/*识别注释并忽略*/
        case '"':
            return         readStr();/*识别字符串常量*/
        }

        /*识别数字*/
        if (Character.isDigit(peek)) {
            int v = 0;
            do {
                v = 10 * v + Character.digit(peek, 10);
                readch();
            } while (Character.isDigit(peek));
            /*如果不是小数点（浮点数），不是科学计数法，不是十六进制，就输出整数*/
            if ((peek != '.')&&(peek!='x')&&(peek!='X')&&(peek!='e')&&(peek!='E')&&(peek!='A')&&(peek!='B')&&(peek!='C')&&(peek!='D')&&(peek!='F'))
                return new Num(v);
            if (peek == '.') {/*如果是小数点的话，就输出浮点数*/
            float x = v;
            float d = 10;
            readch();
            for (;;) {
                if (!Character.isDigit(peek))
                    break;
                x = x + Character.digit(peek, 10) / d;
                d = d * 10;
                readch();
            }
            if(peek=='E'||peek=='e'){/*识别科学计数法浮点数*/
                readch();
                int count=0;
                if(peek=='+'){/*正数*/
                    count=readInt();/*读取整数*/
                    for(;count>0;count--){
                        x*=10;
                    }
                }else if(peek=='-'){/*负数*/
                    count=readInt();/*读取整数*/
                    for(;count>0;count--){
                        x/=10;
                    }
                }
            }
            return new Real(x);/*识别十进制数字*/}
            if (((peek=='x')||(peek=='X'))&&(v==0)) {/*识别十六进制整数*/
                readch();
                do {
                    v = 16 * v + Character.digit(peek, 16);
                    readch();
                } while ((Character.isLetterOrDigit(peek)));

                return new Num(v);
            }

        }


        if (Character.isLetter(peek)) {/*如果识别失败，就开始当作字符串读入（关键字或者标识串）*/
            StringBuffer b = new StringBuffer();
            do {/*得到一个整的分割*/
                b.append(peek);
                readch();
            } while ((Character.isLetterOrDigit(peek))||(peek == '_'));
            String s = b.toString();
            Word w = (Word) words.get(s);
            if (w != null)
                return w;
            w = new Word(s, Tag.ID);/*是一标识符*/
            words.put(s, w);
            return w;
        }
        Token tok = new Token(peek);/*最后，peek中的任意字符都被当作词法单元返回*/
        peek = ' ';
        return tok;
    }



    public void out() {
        System.out.println(words.size());

    }

    public char getPeek() {/*获取当前读到的字符*/
        return peek;
    }

    public void setPeek(char peek) {
        this.peek = peek;
    }

}
